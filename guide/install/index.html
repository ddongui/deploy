# 安装

# 安装本地依赖

npm i ddongui/deploy -D

pnpm add ddongui/deploy -D

# 安装全局依赖

安装全局依赖后可直接使用deploy命令， 自动扫描配置文件

npm i ddongui/deploy -D -g

pnpm add ddongui/deploy -D -g
